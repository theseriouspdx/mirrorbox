## 5. src/state/state-manager.js — Pre-mutation checkpoint (BUG-041)

--- a/src/state/state-manager.js
+++ b/src/state/state-manager.js
@@ add checkpoint() method @@

+  /**
+   * Invariant 13: Pre-mutation checkpoint.
+   * Must be called before any write that mutates project state.
+   * Writes a CHECKPOINT event to the event store and a snapshot to the db.
+   */
+  checkpoint(label = 'pre-mutation') {
+    const snapshot = JSON.stringify(this.getStateSummary());
+    this.db.prepare(
+      'INSERT INTO checkpoints (id, label, snapshot, timestamp) VALUES (?, ?, ?, ?)'
+    ).run(randomUUID(), label, snapshot, Date.now());
+    eventStore.append('CHECKPOINT', 'operator', { label, snapshotSize: snapshot.length });
+    console.error(`[StateManager] Checkpoint written: ${label}`);
+  }
+
+  rollbackToCheckpoint(checkpointId) {
+    const row = this.db.prepare('SELECT snapshot FROM checkpoints WHERE id = ?').get(checkpointId);
+    if (!row) throw new Error(`[StateManager] Checkpoint ${checkpointId} not found.`);
+    const snapshot = JSON.parse(row.snapshot);
+    this.stateSummary = snapshot;
+    eventStore.append('ROLLBACK', 'operator', { checkpointId });
+    console.error(`[StateManager] Rolled back to checkpoint: ${checkpointId}`);
+  }

```python
#!/usr/bin/env python3
"""
MBO Handshake Gate — Fortress 2.0
==================================
Manages physical filesystem permission elevation and Merkle context verification.
"""

import hashlib
import json
import os
import stat
import sys
import time
import uuid
from pathlib import Path
from typing import Optional

# Paths
MBO_ROOT        = Path(__file__).parent.parent
SRC_DIR         = MBO_ROOT / "src"
JOURNAL_DIR     = MBO_ROOT / ".journal"
STATE_FILE      = JOURNAL_DIR / "state.json"
SESSION_LOCK    = JOURNAL_DIR / "session.lock"
AGENTS_MD       = MBO_ROOT / ".dev" / "governance" / "AGENTS.md"

SESSION_TTL_SECONDS = 1800

def _sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    h.update(path.read_bytes())
    return h.hexdigest()

def compute_merkle_root(src_dir: Path) -> str:
    leaves = []
    for filepath in sorted(src_dir.rglob("*.py")):
        relative = str(filepath.relative_to(src_dir))
        content_hash = _sha256_file(filepath)
        leaf = hashlib.sha256(f"{relative}:{content_hash}".encode()).hexdigest()
        leaves.append(leaf)
    if not leaves:
        return hashlib.sha256(b"__mbo_empty_src__").hexdigest()
    layer = leaves
    while len(layer) > 1:
        next_layer = []
        for i in range(0, len(layer), 2):
            left  = layer[i]
            right = layer[i + 1] if i + 1 < len(layer) else left
            next_layer.append(hashlib.sha256(f"{left}{right}".encode()).hexdigest())
        layer = next_layer
    return layer[0]

def load_canonical_state() -> dict:
    if not STATE_FILE.exists():
        _fatal("state.json not found. Run `bin/init_state.py` to baseline.")
    try:
        return json.loads(STATE_FILE.read_text())
    except json.JSONDecodeError as e:
        _fatal(f"state.json is malformed: {e}")

def save_state(merkle_root: str) -> None:
    state = {
        "merkle_root": merkle_root,
        "last_verified": time.time(),
        "last_verified_iso": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
    }
    STATE_FILE.write_text(json.dumps(state, indent=2))

def check_existing_session() -> Optional[dict]:
    if not SESSION_LOCK.exists():
        return None
    try:
        session = json.loads(SESSION_LOCK.read_text())
    except (json.JSONDecodeError, KeyError):
        SESSION_LOCK.unlink(missing_ok=True)
        return None
    if time.time() > session.get("expires_at", 0):
        SESSION_LOCK.unlink(missing_ok=True)
        return None
    return session

def issue_session_token(cell_name: str, merkle_root: str) -> dict:
    token = {
        "token":       str(uuid.uuid4()),
        "cell_scope":  cell_name,
        "merkle_root": merkle_root,
        "issued_at":   time.time(),
        "expires_at":  time.time() + SESSION_TTL_SECONDS,
    }
    SESSION_LOCK.write_text(json.dumps(token, indent=2))
    return token

READONLY  = stat.S_IRUSR | stat.S_IRGRP | stat.S_IROTH
READWRITE = stat.S_IRUSR | stat.S_IWUSR | stat.S_IRGRP | stat.S_IROTH

def lock_src() -> None:
    for item in SRC_DIR.rglob("*"):
        try:
            if item.is_file():
                item.chmod(READONLY)
        except PermissionError:
            pass

def elevate_cell(cell_path: Path) -> None:
    for item in cell_path.rglob("*"):
        try:
            if item.is_file():
                item.chmod(READWRITE)
        except PermissionError as e:
            _fatal(f"Cannot elevate permissions for {item}: {e}")

def enter_hydration_mode(canonical_root: str, computed_root: str) -> None:
    _log("CONTEXT MISMATCH — Entering Hydration Mode.")
    _append_journal_event("HYDRATION_MODE", {
        "canonical_root": canonical_root,
        "computed_root":  computed_root,
        "timestamp":      time.time(),
    })
    sys.exit(1)

def handshake(cell_name: str) -> None:
    existing = check_existing_session()
    if existing:
        _fatal(f"Active session already held for cell '{existing['cell_scope']}'.")
    state = load_canonical_state()
    canonical_root = state.get("merkle_root")
    current_root = compute_merkle_root(SRC_DIR)
    if current_root != canonical_root:
        enter_hydration_mode(canonical_root, current_root)
        return
    cell_path = SRC_DIR / "cells" / cell_name
    if not cell_path.exists():
        _fatal(f"Cell '{cell_name}' does not exist.")
    lock_src()
    elevate_cell(cell_path)
    issue_session_token(cell_name, current_root)
    _append_journal_event("SESSION_OPEN", {"cell_scope": cell_name})

def revoke() -> None:
    session = check_existing_session()
    lock_src()
    SESSION_LOCK.unlink(missing_ok=True)
    if session:
        _append_journal_event("SESSION_REVOKED", {"cell_scope": session.get("cell_scope")})

def status() -> None:
    session = check_existing_session()
    print(f"[SESSION]  {'Active' if session else 'None'}")
    if STATE_FILE.exists():
        state = json.loads(STATE_FILE.read_text())
        print(f"[MERKLE]   {state.get('merkle_root', 'unknown')[:32]}...")

def _log(msg: str) -> None: print(f"[GATE] {msg}")
def _fatal(msg: str) -> None: print(f"[GATE] DENIED: {msg}", file=sys.stderr); sys.exit(1)
def _append_journal_event(event_type: str, payload: dict) -> None:
    with open(JOURNAL_DIR / "audit.log", "a") as f:
        f.write(json.dumps({"event": event_type, "payload": payload}) + "\n")

if __name__ == "__main__":
    if len(sys.argv) < 2: sys.exit(1)
    cmd = sys.argv[1]
    if cmd == "--revoke": revoke()
    elif cmd == "--status": status()
    else: handshake(cmd)
```

### `bin/validator.py`
```python
#!/usr/bin/env python3
"""
MBO Validator — The "Cynic" AST Engine (Fortress 2.0)
"""
import ast
import json
import sys
import time
from dataclasses import dataclass, field
from pathlib import Path
from typing import Optional

MBO_ROOT         = Path(__file__).parent.parent
SRC_DIR          = MBO_ROOT / "src"
CELLS_DIR        = SRC_DIR / "cells"
IMPORTS_ALLOW    = MBO_ROOT / ".dev" / "governance" / "imports.allow"
COMPLEXITY_TOML  = MBO_ROOT / ".dev" / "governance" / "complexity.toml"
JOURNAL_DIR      = MBO_ROOT / ".journal"

DEFAULT_LOC_LIMIT      = 250
DEFAULT_CC_LIMIT       = 10
DEFAULT_NESTING_LIMIT  = 4
DEFAULT_IMPORT_LIMIT   = 5

BRANCH_NODES = (ast.If, ast.For, ast.While, ast.ExceptHandler, ast.With, ast.BoolOp, ast.Match, ast.MatchCase, ast.AsyncFor, ast.AsyncWith)
NESTING_NODES = (ast.If, ast.For, ast.While, ast.With, ast.Try, ast.AsyncFor, ast.AsyncWith, ast.ExceptHandler)

@dataclass
class Violation:
    axis: str; got: int | str; limit: int | str; note: str = ""
    def __str__(self) -> str: return f"[{self.axis}] got={self.got!r} limit={self.limit!r} ({self.note})"

@dataclass
class FileResult:
    filepath: str; violations: list[Violation] = field(default_factory=list); skipped: bool = False; skip_reason: str = ""
    @property
    def passed(self) -> bool: return not self.violations and not self.skipped

@dataclass
class BudgetConfig:
    loc_limit: int = DEFAULT_LOC_LIMIT; cc_limit: int = DEFAULT_CC_LIMIT; nesting_limit: int = DEFAULT_NESTING_LIMIT; import_limit: int = DEFAULT_IMPORT_LIMIT

def load_budget_config() -> BudgetConfig:
    cfg = BudgetConfig()
    if not COMPLEXITY_TOML.exists(): return cfg
    try:
        import tomllib
        data = tomllib.loads(COMPLEXITY_TOML.read_text())
        limits = data.get("limits", {})
        cfg.loc_limit = limits.get("max_loc", cfg.loc_limit)
        cfg.cc_limit = limits.get("max_cc", cfg.cc_limit)
        cfg.nesting_limit = limits.get("max_nesting", cfg.nesting_limit)
    except: pass
    return cfg

def load_import_whitelist() -> Optional[set[str]]:
    if not IMPORTS_ALLOW.exists(): return None
    return {line.strip() for line in IMPORTS_ALLOW.read_text().splitlines() if line.strip() and not line.startswith("#")}

def count_loc(source: str) -> int:
    return sum(1 for line in source.splitlines() if line.strip() and not line.strip().startswith("#"))

def cyclomatic_complexity(tree: ast.AST) -> int:
    return 1 + sum(1 for node in ast.walk(tree) if isinstance(node, BRANCH_NODES))

def max_nesting_depth(tree: ast.AST) -> int:
    def _depth(node: ast.AST, current_depth: int) -> int:
        if isinstance(node, NESTING_NODES): current_depth += 1
        return max([_depth(child, current_depth) for child in ast.iter_child_nodes(node)], default=current_depth)
    return _depth(tree, 0)

def validate_file(filepath: Path, cfg: BudgetConfig, whitelist: Optional[set[str]]) -> FileResult:
    result = FileResult(filepath=str(filepath))
    try: source = filepath.read_text(encoding="utf-8")
    except: result.skipped = True; return result
    try: tree = ast.parse(source)
    except Exception as e: result.violations.append(Violation("SYNTAX_ERROR", str(e), "valid Python")); return result
    loc = count_loc(source)
    if loc > cfg.loc_limit: result.violations.append(Violation("LOC", loc, cfg.loc_limit))
    cc = cyclomatic_complexity(tree)
    if cc > cfg.cc_limit: result.violations.append(Violation("CC", cc, cfg.cc_limit))
    depth = max_nesting_depth(tree)
    if depth > cfg.nesting_limit: result.violations.append(Violation("NESTING", depth, cfg.nesting_limit))
    if whitelist is not None:
        for node in ast.walk(tree):
            if isinstance(node, ast.Import):
                for n in node.names:
                    if n.name.split('.')[0] not in whitelist: result.violations.append(Violation("IMPORT", n.name, "whitelist"))
    return result

def main():
    cfg = load_budget_config(); whitelist = load_import_whitelist(); results = []
    if sys.argv[1] == "--all":
        for cell in CELLS_DIR.iterdir():
            if cell.is_dir(): results.extend([validate_file(f, cfg, whitelist) for f in cell.rglob("*.py")])
    passed = all(r.passed for r in results)
    for r in results:
        if not r.passed: print(f"FAIL: {r.filepath}"); [print(f"  {v}") for v in r.violations]
    sys.exit(0 if passed else 1)

if __name__ == "__main__": main()
```
